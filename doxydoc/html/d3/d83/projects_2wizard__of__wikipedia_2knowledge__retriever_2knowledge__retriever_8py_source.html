215  def tfidf_retriever_act(self, history):
216  """
217  Combines and formats texts retrieved by the TFIDF retriever for the chosen
218  topic, the last thing the wizard said, and the last thing the apprentice said.
219  """
220  # retrieve on chosen topic

227 
228  # retrieve on apprentice
229  apprentice_txts = None
230  if self.retriever_history.get('apprentice'):
231  apprentice_act = {
232  'text': self.retriever_history['apprentice'],
233  'episode_done': True,
234  }
235  self.retriever.observe(apprentice_act)
236  apprentice_txts, apprentice_txts_no_title = self.get_passages(
237  self.retriever.act()
238  )
239 
240  # retrieve on wizard
241  wizard_txts = None
242  if self.retriever_history.get('wizard'):
243  wizard_act = {
244  'text': self.retriever_history['wizard'],
245  'episode_done': True,
246  }
247  self.retriever.observe(wizard_act)
248  wizard_txts, wizard_txts_no_title = self.get_passages(self.retriever.act())
249 
250  # combine everything
251  combined_txt = ''
252  combined_txt_no_title = ''
253  if chosen_topic_txts:
254  combined_txt += chosen_topic_txts
255  combined_txt_no_title += chosen_topic_txts_no_title
256  if wizard_txts:
257  combined_txt += '\n' + wizard_txts
258  combined_txt_no_title += '\n' + wizard_txts_no_title
259  if apprentice_txts:
260  combined_txt += '\n' + apprentice_txts
261  combined_txt_no_title += '\n' + apprentice_txts_no_title
262

300  knowledge, knowledge_no_title = self.tfidf_retriever_act(self.retriever_history)
