107 def add_person_tokens(responses, first_speaker=None, last_speaker=1):
108  """Converts a list of responses into a single tag-separated string
109  Args:
110  responses: list of responses (strings)
111  first_speaker: either 1 or 2; the owner of the first response
112  last_speaker: either 1 or 2; the owner of the last response
113  NOTE: if first_speaker is provided, it overrides last_speaker
114  Output:
115  text: the concatenated text
116 
117  e.g.,
118  responses = ["How are you?", "I'm doing fine!", "I'm glad to hear it!"]
119  result = add_person_tokens(responses)
120  result: "__p1__ How are you? __p2__ I'm doing fine! __p1__ I'm glad to
121  hear it!"
122  """
123  if first_speaker is None:
124  first_speaker = (last_speaker + len(responses)) % 2 + 1
125  speaker = first_speaker
126  text = ''
127  for response in responses:
128  tag = f"__p{speaker}__"
129  text += ' ' + tag + ' ' + response
130  speaker = 1 if speaker == 2 else 2
131  return text.strip()

161 def episode_to_examples(episode, histsz):
162  """
163  Converts an episode (list of Parleys) into self-feeding compatible examples.
164 
165  WARNING: we no longer require a histz when making a self-feeding file. Shortening of
166  the history is typically done in the teacher file or in interactive mode.
167  """

171  # Update memories and history
172  # memories.extend(parley.memories)
173  history.append(parley.context)
174 
175  # Concatenate history and add speaker tokens as necessary
176  # if history_size == 1, the bot (p2) only sees the immediately
177  # preceding utterance (the prompt from the human, p1).
178  if histsz < 0:
179  utterances = history
180  context = add_person_tokens(utterances, last_speaker=1)
181  elif histsz == 0:
182  context = '__null__'
183  else:
184  utterances = history[-histsz:]
185  context = add_person_tokens(utterances, last_speaker=1)
