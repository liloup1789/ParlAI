84  def _get_new_topic(self):

88  letters = list(string.ascii_uppercase)[: self.num_topics]

103  done = True
104  else:

127  def parley(self):
128  """
129  Loop between wizard and apprentice.
130 
131  Adds knowledge to the wizard observations. Assumes that the model agent is the
132  wizard model.
133  """
134 
135  if self.cnt == 0:
136  self.topic = self._get_new_topic()
137  self.acts = [None, None]
138  self.human_first = random.choice([0, 1])
139 
140  # possibly get human act first
141  if self.cnt == 0 and not self.human_first:
142  self.acts[0] = act = Message({'text': '', 'episode_done': False})
143  act = self.acts[0]
144  else:
145  self.acts[0] = self.human_agent.act()
146  act = deepcopy(self.acts[0])
147 
148  # model agent observe
149  if self.cnt == 0 and self.topic != NO_TOPIC:
150  # add the chosen_topic to the message
151  act['chosen_topic'] = self.topic
152  act.force_set('text', '\n'.join([self.topic, act.get('text', 'hi')]))
153 
154  # add knowledge to the model observation
155  act = self._add_knowledge_to_act(act)
156 
157  # model observes knowledge and human (apprentice) act
158  self.model_agent.observe(validate(act))
159 
160  # model agent act
161  self.acts[1] = self.model_agent.act()
162 
163  # add the model reply to the knowledge retriever's dialogue history
164  self.knowledge_agent.observe(self.acts[1], actor_id='wizard')
165 
166  # human (apprentice) agent observes model act
167  self.human_agent.observe(validate(self.acts[1]))
168 
169  self.update_counters()
170  self.cnt += 1
171 
172  if self.episode_done():
173  print('[ CHAT DONE ]')
174  print('\n[ Preparing new chat... ]\n')
175  self.cnt = 0
176  self.model_agent.reset()
