397  def act(self, timeout=None, blocking=True):
398  """
399  Sends a message to other agents in the world.
400 
401  If blocking, this will wait for the message to come in so it can be sent.
402  Otherwise it will return None.
403  """
404  if not blocking:
405  # if this is the first act since last sent message start timing
406  if self.message_request_time is None:
407  self.request_message()
408  self.message_request_time = time.time()
409 
410  # If checking timeouts
411  if timeout:
412  # If time is exceeded, timeout
413  if time.time() - self.message_request_time > timeout:
414  raise AgentTimeoutError(timeout, self.worker_id, self.assignment_id)
415 
416  # Get a new message, if it's not None reset the timeout
417  msg = self.get_new_act_message()
418  if msg is not None and self.message_request_time is not None:
419  self.message_request_time = None
420  return msg
421  else:
422  self.request_message()
423  self.message_request_time = time.time()
424 
425  # Timeout in seconds, after which the HIT is expired automatically
426  if timeout:
427  start_time = time.time()
428 
429  # Wait for agent's new message
430  while True:
431  msg = self.get_new_act_message()
432  self.message_request_time = None
433  if msg is not None:
434  return msg
435 
436  # Check if the Turker waited too long to respond
437  if timeout:
438  current_time = time.time()
439  if (current_time - start_time) > timeout:
440  self.message_request_time = None
441  raise AgentTimeoutError(
442  timeout, self.worker_id, self.assignment_id
443  )
444  time.sleep(shared_utils.THREAD_SHORT_SLEEP)
