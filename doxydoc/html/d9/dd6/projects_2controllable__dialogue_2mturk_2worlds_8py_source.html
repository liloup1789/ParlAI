393  def format_model_reply(self, text):
394  switch_list = [(' .', '.'), (' ,', ','), (' ?', '?'), (' !', '!'), (" ' ", "'")]
395  # add the spaces so that
396  new_text = text.lower()
397 
398  # normalize in case of human:
399  for new, old in switch_list:
400  new_text = new_text.replace(old, new).replace(' ', ' ')
401 
402  # split on punctuation to find sentence boundaries
403  # capitalize stuff
404  tokens = new_text.split(' ')
405  for i in range(len(tokens)):
406  if i == 0:
407  tokens[i] = uppercase(tokens[i])
408  elif tokens[i] in ('i', "i'm", "i've", "i'll", "i'd"):
409  tokens[i] = uppercase(tokens[i])
410  elif tokens[i] in '?.!' and i < len(tokens) - 1:
411  tokens[i + 1] = uppercase(tokens[i + 1])
412  new_text = ' '.join(tokens)
413  new_text = ' ' + new_text + ' '
414 
415  for tup in switch_list:
416  new_text = new_text.replace(tup[0], tup[1])
417 
418  # get rid of surrounding whitespace
419  new_text = new_text.strip()
420  new_text = new_text.replace(' ', ' ')
421
