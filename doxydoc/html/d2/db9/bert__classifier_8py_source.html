173  def score(self, batch):
174  """
175  Score the batch.
176  """
177  segment_idx = (batch.text_vec * 0).long()
178  if self.sep_last_utt:
179  batch_len = batch.text_vec.size(1)
180  # find where [SEP] token is
181  seps = (batch.text_vec == self.dict.end_idx).nonzero()
182  if len(seps) > 0:
183  for row in seps:
184  # set last utterance to segment 1
185  segment_idx[row[0], list(range(row[1], batch_len))] = 1
186  else:
187  # only one utterance: everything after [CLS] token
188  # should be segment 1
189  segment_idx = (batch.text_vec != self.dict.start_idx).long()
190  mask = (batch.text_vec != self.NULL_IDX).long()
191  token_idx = batch.text_vec * mask
192  return self.model(token_idx, segment_idx, mask)
