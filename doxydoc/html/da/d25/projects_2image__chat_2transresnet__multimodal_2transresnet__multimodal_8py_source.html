115  print("Extracting cand encodings")
116  self.model.eval()
117  pbar = tqdm.tqdm(
118  total=len(self.fixed_cands),
119  unit="cand",
120  unit_scale=True,
121  desc="Extracting candidate encodings",

123  fixed_cands_enc = []
124  for _, batch in enumerate(
125  [

127  for i in range(0, len(self.fixed_cands) - 50, 50)
128  ]

131  fixed_cands_enc.append(embedding)
132  pbar.update(50)
133  self.fixed_cands_enc = torch.cat(fixed_cands_enc, 0)
134  torch.save(self.fixed_cands_enc, cands_enc_file)

191  def eval_step(self, valid_obs, image_feats, personalities, dialogue_histories):
192  """
193  Model eval step.
194 
195  :param valid_obs:
196  list of valid observations
197 
198  :param image_feats:
199  list of image features, one per example
200 
201  :param personalities:
202  list of personalities, one per example
203 
204  :param dialogue_histories:
205  list of dialogue histories, one per example
206 
207  :return:
208  the total loss, number of correct examples,
209  the ranked position of each correct caption,
210  and the ranked lists of candidates (one per example)
211  """
212  self.model.eval()
213  med_rank = None
214  chosen_responses = None
215  candidates_encoded = None
216 
217  if self.fixed_cands is not None:
218  candidates_encoded = self.fixed_cands_enc
219  candidates = self.fixed_cands
220  else:
221  candidates = [v["label_candidates"] for v in valid_obs]
222  chosen_responses = self.model.choose_best_response(
223  image_feats,
224  personalities,
225  dialogue_histories,
226  candidates,
227  candidates_encoded=candidates_encoded,
228  k=-1 if self.fixed_cands is None else 100,
229  batchsize=len(valid_obs),
230  )
231  loss = -1
232  if self.fixed_cands is not None:
233  num_correct = 0
234  else:
235  labels = [v.get("eval_labels") for v in valid_obs]
236  equality_list = [
237  1 if chosen_responses[i][0] in labels[i] else 0
238  for i in range(len(labels))
239  ]
240  # calculate med ranks
241  med_rank = []
242  for i, e_list in enumerate(chosen_responses):
243  lowest_rank = len(e_list) + 1
244  for c in labels[i]:
245  lowest_rank = min(lowest_rank, e_list.index(c) + 1)
246  med_rank.append(lowest_rank)
247  num_correct = sum(equality_list)
248 
249  return loss, num_correct, med_rank, chosen_responses

291  def extract_texts(self, obs):
292  """
293  Extract the personalities and dialogue histories from observations.
294 
295  Additionally determine which dialogue round we are in.
296 
297  Note that this function assumes that the personality is the
298  last line of the `text` field in the observation.
299 
300  :param obs:
301  list of observations
302 
303  :return:
304  a list of personalities, a list of dialogue histories, and the
305  current dialogue round (either first, second, or third+)
306  """
307  splits = [v.get("text").split("\n") for v in obs]
308  if self.personality_override:
309  splits = [s + [self.personality_override] for s in splits]
310  personalities = [t[-1] for t in splits]
311  dialogue_histories = None
312  dialogue_round = "first_round"
313  if len(splits[0]) >= 2:
314  dialogue_round = "second_round" if len(splits[0]) == 2 else "third_round+"
315  dialogue_histories = ["\n".join(t[:-1]) for t in splits]
316 
317  return personalities, dialogue_histories, dialogue_round

364  def receive_metrics(self, metrics_dict):
365  """
366  Receive the metrics from validation.
367 
368  Unfreeze text encoder weights after a certain number of rounds without improvement.
369 
370  Override to account for different dialogue rounds.

372  :param metrics_dict:
373  the metrics dictionary
374  """

377  if self.freeze_patience != -1 and self.is_frozen:
378  m_key = "hits@1/100"

382  ]

384  if m > self.freeze_best_metric:
385  self.freeze_impatience = 0

387  print("performance not good enough to unfreeze the model.")
