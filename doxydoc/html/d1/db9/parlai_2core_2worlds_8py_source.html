6 """
7 Worlds are the basic environments which define how agents interact with one another.
8 
9  ``World(object)`` provides a generic parent class, including ``__enter__``
10  and ``__exit__`` statements which allow you to guarantee that the shutdown
11  method is called.
12 
13  ``DialogPartnerWorld(World)`` provides a two-agent turn-based dialog setting.
14 
15  ``MultiAgentDialogWorld(World)`` provides a multi-agent setting.
16 
17  ``MultiWorld(World)`` creates a set of environments (worlds) for the same agent
18  to multitask over, a different environment will be chosen per episode.
19 
20  ``HogwildWorld(World)`` is a container that creates another world within itself for
21  every thread, in order to have separate simulated environments for each one.
22  Each world gets its own agents initialized using the ``share()`` parameters
23  from the original agents.
24 
25  ``BatchWorld(World)`` is a container for doing minibatch training over a world by
26  collecting batches of N copies of the environment (each with different state).
27 
28 
29 All worlds are initialized with the following parameters:
30 
31  ``opt`` -- contains any options needed to set up the agent. This generally contains
32  all command-line arguments recognized from core.params, as well as other
33  options that might be set through the framework to enable certain modes.
34  ``agents`` -- the set of agents that should be attached to the world,
35  e.g. for DialogPartnerWorld this could be the teacher (that defines the
36  task/dataset) and the learner agent. This is ignored in the case of
37  sharing, and the shared parameter is used instead to initalize agents.
38  ``shared`` (optional) -- if not None, contains any shared data used to construct
39  this particular instantiation of the world. This data might have been
40  initialized by another world, so that different agents can share the same
41  data (possibly in different Processes).
42 """

529  """
530  For each agent: act, execute and observe actions in world.
531  """
532  acts = self.acts
533  for index, agent in enumerate(self.agents):
534  # The agent acts.
535  acts[index] = agent.act()

546 class MultiWorld(World):
547  """
548  Container for multiple worlds.
549 
550  Container for a set of worlds where each world gets a turn in a round-robin fashion.
551  The same user_agents are placed in each, though each world may contain additional
552  agents according to the task that world represents.
553  """
554 
555  def __init__(self, opt: Opt, agents=None, shared=None, default_world=None):
556  super().__init__(opt)
557  self.worlds: List[World] = []
558  for index, k in enumerate(opt['task'].split(',')):
559  k = k.strip()
560  if k:
561  opt_singletask = copy.deepcopy(opt)
562  opt_singletask['task'] = k
563  if shared:
564  # Create worlds based on shared data.
565  s = shared['worlds'][index]
566  self.worlds.append(s['world_class'](s['opt'], None, s))
567  else:
568  # Agents are already specified.
569  self.worlds.append(
570  create_task_world(
571  opt_singletask, agents, default_world=default_world
572  )
573  )
574  self.world_idx = -1
575  self.new_world = True
576  self.parleys = -1
577  self.random = opt.get('datatype', None) == 'train'
578  # Make multi-task task probabilities.
579  self.cum_task_weights = [1] * len(self.worlds)
580  self.task_choices = range(len(self.worlds))
581  weights = self.opt.get('multitask_weights', [1])
582  sum = 0
583  for i in self.task_choices:
584  if len(weights) > i:
585  weight = weights[i]
586  else:
587  weight = 1
588  self.cum_task_weights[i] = weight + sum

652  def parley_init(self):
653  """
654  Update the current subworld.
655 
656  If we are in the middle of an episode, keep the same world and finish this
657  episode. If we have finished this episode, pick a new world (either in a random
658  or round-robin fashion).
659  """
660  self.parleys = self.parleys + 1
661  if self.world_idx >= 0 and self.worlds[self.world_idx].episode_done():
662  self.new_world = True
663  if self.new_world:
664  self.new_world = False
665  self.parleys = 0
666  if self.random:
667  # select random world
668  self.world_idx = random.choices(
669  self.task_choices, cum_weights=self.cum_task_weights
670  )[0]
671  else:
672  # do at most one full loop looking for unfinished world
673  for _ in range(len(self.worlds)):
674  self.world_idx = (self.world_idx + 1) % len(self.worlds)
675  if not self.worlds[self.world_idx].epoch_done():
676  # if this world has examples ready, break
677  break

731 def _override_opts_in_shared(table, overrides):
732  """
733  Override all shared dicts.
734 
735  Looks recursively for ``opt`` dictionaries within shared dict and overrides any key-
736  value pairs with pairs from the overrides dict.
737  """
738  if 'opt' in table:
739  # change values if an 'opt' dict is available
740  for k, v in overrides.items():
741  table['opt'][k] = v
742  for k, v in table.items():
743  # look for sub-dictionaries which also might contain an 'opt' dict
744  if type(v) == dict and k != 'opt' and 'opt' in v:
745  _override_opts_in_shared(v, overrides)
746  elif type(v) == list:
747  for item in v:
748  if type(item) == dict and 'opt' in item:
749  # if this is a list of agent shared dicts, we want to iterate
750  _override_opts_in_shared(item, overrides)
751  else:
752  # if this is e.g. list of candidate strings, stop right away
753  break
754  return table

757 class BatchWorld(World):
758  """
759  BatchWorld contains many copies of the same world.
760 
761  Create a separate world for each item in the batch, sharing
762  the parameters for each.
763 
764  The underlying world(s) it is batching can be either
765  ``DialogPartnerWorld``, ``MultiAgentWorld``, ``ExecutableWorld`` or
766  ``MultiWorld``.
767  """
768 
769  def __init__(self, opt: Opt, world):
770  super().__init__(opt)
771  self.opt = opt
772  self.random = opt.get('datatype', None) == 'train'
773  self.world = world
774  self.worlds: List[World] = []
775  for i in range(opt['batchsize']):
776  # make sure that any opt dicts in shared have batchindex set to i
777  # this lets all shared agents know which batchindex they have,
778  # which is needed for ordered data (esp valid/test sets)
779  shared = world.share()

991  def __init__(self, tid, opt: Opt, shared, sync):
992  self.numthreads = opt['numthreads']
993  opt = copy.deepcopy(opt)
994  opt['numthreads'] = 1 # don't let threads create more threads!
995  self.opt = opt

1004  def run(self):
1005  """
1006  Run a parley loop.
1007 
1008  Runs normal parley loop for as many examples as this thread can get ahold of via
1009  the semaphore ``queued_sem``.
1010  """
1011  world = self.shared['world_class'](self.opt, None, self.shared)
1012  if self.opt.get('batchsize', 1) > 1:
1013  world = BatchWorld(self.opt, world)
1014  self.sync['threads_sem'].release()
1015  with world:
1016  while True:
1017  if self.sync['term_flag'].value:
1018  break # time to close
1019  self.sync['queued_sem'].acquire()
1020  self.sync['threads_sem'].release()
1021 
1022  # check if you need to reset before moving on
1023  if self.sync['epoch_done_ctr'].value < 0:
1024  with self.sync['epoch_done_ctr'].get_lock():
1025  # increment the number of finished threads
1026  self.sync['epoch_done_ctr'].value += 1
1027  if self.sync['epoch_done_ctr'].value == 0:
1028  # make sure reset sem is clean
1029  for _ in range(self.numthreads):
1030  self.sync['reset_sem'].acquire(block=False)
1031  world.reset() # keep lock for this!
1032 
1033  while self.sync['epoch_done_ctr'].value < 0:
1034  # only move forward once other threads have finished reset
1035  time.sleep(0.1)
1036 
1037  # process an example or wait for reset
1038  if not world.epoch_done() or self.opt.get('datatype').startswith(
1039  'train', False
1040  ):
1041  # do one example if any available
1042  world.parley()
1043  with self.sync['total_parleys'].get_lock():
1044  self.sync['total_parleys'].value += 1
1045  else:
1046  # during valid/test, we stop parleying once at end of epoch
1047  with self.sync['epoch_done_ctr'].get_lock():
1048  # increment the number of finished threads
1049  self.sync['epoch_done_ctr'].value += 1
1050  # send control back to main thread
1051  self.sync['threads_sem'].release()
1052  # we didn't process anything
1053  self.sync['queued_sem'].release()
1054  # wait for reset signal
1055  self.sync['reset_sem'].acquire()

1058 class HogwildWorld(World):
1059  """
1060  Creates a separate world for each thread (process).
1061 
1062  Maintains a few shared objects to keep track of state:
1063 
1064  - A Semaphore which represents queued examples to be processed. Every call
1065  of parley increments this counter; every time a Process claims an
1066  example, it decrements this counter.
1067 
1068  - A Condition variable which notifies when there are no more queued
1069  examples.
1070 
1071  - A boolean Value which represents whether the inner worlds should shutdown.
1072 
1073  - An integer Value which contains the number of unprocessed examples queued
1074  (acquiring the semaphore only claims them--this counter is decremented
1075  once the processing is complete).
1076  """
1077 
1078  def __init__(self, opt: Opt, world):
1079  super().__init__(opt)
1080  self.inner_world = world
1081  self.numthreads = opt['numthreads']
1082 
1083  self.sync: Dict[str, Any] = { # syncronization primitives
1084  # semaphores for counting queued examples
1085  'queued_sem': Semaphore(0), # counts num exs to be processed
1086  'threads_sem': Semaphore(0), # counts threads
1087  'reset_sem': Semaphore(0), # allows threads to reset
1088  # flags for communicating with threads
1089  'reset_flag': Value('b', False), # threads should reset
1090  'term_flag': Value('b', False), # threads should terminate
1091  # counters
1092  'epoch_done_ctr': Value('i', 0), # number of done threads
1093  'total_parleys': Value('l', 0), # number of parleys in threads
1094  }
1095 
1096  self.threads: List[HogwildProcess] = []
1097  for i in range(self.numthreads):
1098  self.threads.append(HogwildProcess(i, opt, world.share(), self.sync))
1099  time.sleep(0.05) # delay can help prevent deadlock in thread launches
1100  for t in self.threads:
1101  t.start()
1102 
1103  for _ in self.threads:
1104  # wait for threads to launch
1105  # this makes sure that no threads get examples before all are set up
1106  # otherwise they might reset one another after processing some exs
1107  self.sync['threads_sem'].acquire() # type: ignore
1108 
1109  print(f'[ {self.numthreads} threads initialized ]')
