127  def get(self, episode_idx: int, entry_idx: int = 0) -> dict:
128  """
129  Override to handle corrupt images and multi-reference labels.
130  """
131  entry_idx *= 2
132 
133  if episode_idx >= len(self.data):
134  data = self.data[episode_idx % len(self.data)]
135  entry_idx += 1
136  else:
137  data = self.data[episode_idx]
138  image_id = data[self.image_id_key]
139  if data[self.image_id_key] not in self.valid_image_ids:
140  data[self.image_id_key] = self.blank_image_id
141  image_features = self.get_image_features(data)
142 
143  conversation = [data['context'], data['question'], data['response']]
144  labels = [conversation[entry_idx]]
145  if self.multi_ref and entry_idx != 0:
146  key = 'questions' if entry_idx == 1 else 'responses'
147  labels = data[f'multiref_{key}'].split('***')
148  text = '' if entry_idx == 0 else conversation[entry_idx - 1]
149  episode_done = entry_idx >= len(conversation) - 2
150 
151  action = {
152  'text': text,
153  'image_id': image_id,
154  'episode_done': episode_done,
155  'image': image_features,
156  'labels': labels,
157  }
158 
159  return action

184  with open(dp, newline='\n') as csv_file:
