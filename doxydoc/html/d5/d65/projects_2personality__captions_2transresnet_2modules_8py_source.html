367  def choose_best_caption(
368  self, image_features, personalities, candidates, candidates_encoded=None, k=1
369  ):
370  """
371  Choose the best caption for each example.
372 
373  :param image_features:
374  list of tensors of image features
375  :param personalities:
376  list of personalities
377  :param candidates:
378  list of candidates, one set per example
379  :param candidates_encoded:
380  optional; if specified, a fixed set of encoded candidates that is
381  used for each example
382  :param k:
383  number of ranked candidates to return. if < 1, we return the ranks
384  of all candidates in the set.
385 
386  :return:
387  a set of ranked candidates for each example
388  """
389  self.eval()
390  context_encoded, _ = self.forward(image_features, personalities, None)
391  context_encoded = context_encoded.detach()
392  one_cand_set = True
393  if candidates_encoded is None:
394  one_cand_set = False
395  candidates_encoded = [
396  self.forward(None, None, c)[1].detach() for c in candidates
397  ]
398  chosen = []
399  for img_index in range(len(context_encoded)):
400  context_encoding = context_encoded[img_index : img_index + 1, :]
401  scores = torch.mm(
402  candidates_encoded[img_index]
403  if not one_cand_set
404  else candidates_encoded,
405  context_encoding.transpose(0, 1),
406  )
407  if k >= 1:
408  _, index_top = torch.topk(scores, k, dim=0)
409  else:
410  _, index_top = torch.topk(scores, scores.size(0), dim=0)
411  chosen.append(
412  [
413  candidates[img_index][idx] if not one_cand_set else candidates[idx]
414  for idx in index_top.unsqueeze(1)
415  ]
416  )
417 
418  return chosen

420  def eval_batch_of_100(self, context_encoded, captions_encoded):
421  """
422  Evaluate a batch of 100 examples.
423 
424  The captions of the other examples are used as negatives.
425 
426  :param context_encoded:
427  the encoded context
428  :param captions_encoded:
429  the encoded captions
430 
431  :return:
432  the total loss, the total number of correct examples, and the
433  total number of examples evaluated.
434  """
435  total_loss = 0
436  total_ok = 0
437  num_examples = 0
438  for i in range(0, len(context_encoded), 100):
439  if i + 100 > len(context_encoded):
440  break
441  num_examples += 100
442  loss, num_correct = self.evaluate_one_batch(
443  context_encoded[i : i + 100, :], captions_encoded[i : i + 100, :]
444  )
445  total_loss += loss.data.cpu().item()
446  total_ok += num_correct.data.cpu().item()
447  return total_loss, total_ok, num_examples

449  def evaluate_one_batch(self, context_encoded, captions_encoded, during_train=False):
450  """
451  Compute loss - and number of correct examples - for one batch.
452 
453  :param context_encoded:
454  the encoded context
455  :param captions_encoded:
456  the encoded captions
457  :param during_train:
458  true if training, else False
459 
460  :return:
461  the batch loss and the number of correct examples
462  """
463  if not during_train:
464  self.zero_grad()
465  self.eval()
466  dot_products = context_encoded.mm(captions_encoded.t())
467  log_prob = torch.nn.functional.log_softmax(dot_products, dim=1)
468  targets = torch.arange(0, len(context_encoded), dtype=torch.long)
469  if self.use_cuda:
470  targets = targets.cuda()
471  loss = torch.nn.functional.nll_loss(log_prob, targets)
472  num_correct = (log_prob.max(dim=1)[1] == targets).float().sum()
473  return loss, num_correct

518  def captions_to_tensor(self, captions):
519  """
520  Tokenize a list of sentences into a 2D float tensor.
521 
522  :param captions:
523  list of sentences to tokenize
524 
525  :return:
526  a (batchsize X truncate_length) tensor representation of the captions,
527  and a similarly sized mask tensor
528  """
529  max_length = self.truncate_length
530  indexes = []
531  for c in captions:
532  vec = self.dictionary.txt2vec(c)
533  if len(vec) > max_length:
534  vec = vec[:max_length]
535  indexes.append(self.dictionary.txt2vec(c))
536  longest = max([len(v) for v in indexes])
537  res = torch.LongTensor(len(captions), longest).fill_(
538  self.dictionary.tok2ind[self.dictionary.null_token]
539  )
540  mask = torch.FloatTensor(len(captions), longest).fill_(0)
541  for i, inds in enumerate(indexes):
542  res[i, 0 : len(inds)] = torch.LongTensor(inds)
543  mask[i, 0 : len(inds)] = torch.FloatTensor([1] * len(inds))
544  if self.use_cuda:
545  res = res.cuda()
546  mask = mask.cuda()
