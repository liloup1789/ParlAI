116  """
117  Load global fixed set of candidate labels that the teacher provides.

119  Every example will include these as candidates. The true labels for a specific

123  return None

132  if cnt == 1 and line[0:2] == '1 ':

136  cands_are_replies = True

270  def log(self, done, total, report=None):
271  """
272  Log report, time elapsed, and percentage progress towards goal.
273 
274  :param done: number of examples completed so far
275  :param total: total number of elements to be completed. if total > 0,
276  calculates the time remaining and percentage complete.
277  :param report: dict of pairs to log
278 
279  :returns: tuple log string, log dict

294  z = '%.2f' % (100 * log['%done'])

297  for k, v in report.items():

416  def valid(obs):
417  # check if this is an example our model should actually process
418  return 'text' in obs and len(obs['text']) > 0
419 
420  try:
421  # valid examples and their indices
422  valid_inds, exs = zip(
423  *[(i, ex) for i, ex in enumerate(observations) if valid(ex)]
424  )
425  except ValueError:
426  # zero examples to process in this batch, so zip failed to unpack
427  return None, None, None, None, None, None
428 
429  # `x` text is already tokenized and truncated
430  # sort by length so we can use pack_padded
431  if any(['text2vec' in ex for ex in exs]):
432  parsed_x = [ex['text2vec'] for ex in exs]
433  else:
434  parsed_x = [dictionary.txt2vec(ex['text']) for ex in exs]
435 
436  if len(parsed_x) > 0 and not isinstance(parsed_x[0], deque):
437  if dq:
438  parsed_x = [deque(x, maxlen=truncate) for x in parsed_x]
439  elif truncate is not None and truncate > 0:
440  parsed_x = [x[-truncate:] for x in parsed_x]
441 
442  x_lens = [len(x) for x in parsed_x]
443  ind_sorted = sorted(range(len(x_lens)), key=lambda k: -x_lens[k])
444 
445  exs = [exs[k] for k in ind_sorted]
446  valid_inds = [valid_inds[k] for k in ind_sorted]
447  parsed_x = [parsed_x[k] for k in ind_sorted]
448  end_idxs = [x_lens[k] for k in ind_sorted]
449

451  labels_avail = any(['labels' in ex for ex in exs])

503  for y in parsed_y
504  ]
505  else:

507  y if len(y) == max_y_len else y + [null_idx] * (max_y_len - len(y))

587 def _ellipse(lst: List[str], max_display: int = 5, sep: str = '|') -> str:
588  """
589  Like join, but possibly inserts an ellipsis.

591  :param lst: The list to join on

618  def _token_losses_line(

620  ) -> Optional[str]:

622  Displays the loss associated with each token. Can be used for debugging
623  generative models.
624 
625  See TorchGeneratorAgent._construct_token_losses for an example implementation.
626  """
627  key = 'token_losses'
628  token_losses = msg.get(key, None)
629  if key in ignore_fields or not token_losses:
630  return None
631  # Reduce losses to 4 significant figures
632  formatted_tl = ' | '.join(
633  [f"{tl[0]} {float('{:.4g}'.format(tl[1]))}" for tl in token_losses]
634  )
635  return f'{space}[{key}]: {formatted_tl}'
636 
637  lines = []
638  episode_done = False
639  ignore_fields_ = ignore_fields.split(',')
640  for index, msg in enumerate(msgs):
641  if msg is None or (index == 1 and 'agent_reply' in ignore_fields_):
642  # We only display the first agent (typically the teacher) if we
643  # are ignoring the agent reply.
644  continue
645  if msg.get('episode_done'):
646  episode_done = True
647  # Possibly indent the text (for the second speaker, if two).
648  space = ''
649  if len(msgs) == 2 and index == 1:
650  space = ' '
651  # Only display rewards !=0 as they are confusing in non-RL tasks.
652  if msg.get('reward', 0) != 0:
653  lines.append(space + '[reward: {r}]'.format(r=msg['reward']))
654  for key in msg:
655  if key not in DISPLAY_MESSAGE_DEFAULT_FIELDS and key not in ignore_fields_:
656  if type(msg[key]) is list:
657  line = '[' + key + ']:\n ' + _ellipse(msg[key], sep='\n ')
658  else:
659  line = '[' + key + ']: ' + clip_text(str(msg.get(key)), max_len)
660  lines.append(space + line)
661  if type(msg.get('image')) in [str, torch.Tensor]:
662  lines.append(f'[ image ]: {msg["image"]}')
663  if msg.get('text', ''):
664  text = clip_text(msg['text'], max_len)
665  ID = '[' + msg['id'] + ']: ' if 'id' in msg else ''
666  lines.append(space + ID + text)
667  for field in {'labels', 'eval_labels', 'label_candidates', 'text_candidates'}:
668  if msg.get(field) and field not in ignore_fields_:
669  lines.append('{}[{}: {}]'.format(space, field, _ellipse(msg[field])))
670  # Handling this separately since we need to clean up the raw output before displaying.
671  token_loss_line = _token_losses_line(msg, ignore_fields_, space)
672  if token_loss_line:
673  lines.append(token_loss_line)
674 
675  if episode_done:
676  lines.append('- - - - - - - - - - - - - - - - - - - - -')
677

681 def str_to_msg(txt, ignore_fields=''):
682  """
683  Convert formatted string to ParlAI message dict.
684 
685  :param txt:
686  formatted string to convert. String format is tab-separated fields,
687  with colon separating field name and contents.
688  :param ignore_fields:
689  (default '') comma-separated field names to not
690  include in the msg dict even if they're in the string.
691  """
692 
693  def tostr(txt):
694  txt = str(txt)
695  txt = txt.replace('\\t', '\t')
696  txt = txt.replace('\\n', '\n')
697  txt = txt.replace('__PIPE__', '|')
698  return txt
699 
700  def tolist(txt):
701  vals = txt.split('|')
702  for v in vals:
703  v = tostr(v)
704  return vals
705 
706  def convert(key, value):
707  if key == 'text' or key == 'id':
708  return tostr(value)
709  elif (
710  key == 'label_candidates'
711  or key == 'labels'
712  or key == 'eval_labels'
713  or key == 'text_candidates'
714  ):
715  return tolist(value)
716  elif key == 'episode_done':
717  return bool(value)
718  else:
719  return tostr(value)
720 
721  if txt == '' or txt is None:
722  return None
723 
724  msg = {}
725  for t in txt.split('\t'):
726  ind = t.find(':')
727  key = t[:ind]
728  value = t[ind + 1 :]
729  if key not in ignore_fields.split(','):
730  msg[key] = convert(key, value)
731  msg['episode_done'] = msg.get('episode_done', False)
732  return Message(msg)
