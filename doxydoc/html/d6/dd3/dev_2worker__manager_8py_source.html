206  def load_disconnects(self):
207  """
208  Load disconnects from file, populate the disconnects field for any worker_id
209  that has disconnects in the list.
210 
211  Any disconnect that occurred longer ago than the disconnect persist length is
212  ignored
213  """
214  self.disconnects = []
215  # Load disconnects from file
216  file_path = os.path.join(parent_dir, DISCONNECT_FILE_NAME)
217  compare_time = time.time()
218  if os.path.exists(file_path):
219  with open(file_path, 'rb') as f:
220  old_disconnects = pickle.load(f)
221  self.disconnects = [
222  d
223  for d in old_disconnects
224  if (compare_time - d['time']) < DISCONNECT_PERSIST_LENGTH
225  ]
226  # Initialize worker states with proper number of disconnects
227  for disconnect in self.disconnects:
228  worker_id = disconnect['id']
229  if worker_id not in self.mturk_workers:
230  # add this worker to the worker state
231  self.mturk_workers[worker_id] = WorkerState(worker_id)
232  self.mturk_workers[worker_id].disconnects += 1

245  def handle_agent_disconnect(self, worker_id, assignment_id, partner_callback):
246  """
247  Handles a disconnect by the given worker, calls partner_callback on all of the
248  conversation partners of that worker.
249  """
250  agent = self._get_agent(worker_id, assignment_id)
251  if agent is not None:
252  # Disconnect in conversation is not workable
253  agent.set_status(AssignState.STATUS_DISCONNECT)
254  # in conversation, inform others about disconnect
255  conversation_id = agent.conversation_id
256  if conversation_id in self.conv_to_agent:
257  conv_participants = self.conv_to_agent[conversation_id]
258  if agent in conv_participants:
259  for other_agent in conv_participants:
260  if agent.assignment_id != other_agent.assignment_id:
261  partner_callback(other_agent)
262  if len(conv_participants) > 1:
263  # The user disconnected from inside a conversation with
264  # another turker, record this as bad behavior
265  self.handle_bad_disconnect(worker_id)

267  def handle_bad_disconnect(self, worker_id):
268  """
269  Update the number of bad disconnects for the given worker, block them if they've
270  exceeded the disconnect limit.
271  """
272  if not self.is_sandbox:
273  self.mturk_workers[worker_id].disconnects += 1
274  self.disconnects.append({'time': time.time(), 'id': worker_id})
275  if self.mturk_workers[worker_id].disconnects > MAX_DISCONNECTS:
276  if self.opt['hard_block']:
277  text = (
278  'This worker has repeatedly disconnected from these '
279  'tasks, which require constant connection to complete '
280  'properly as they involve interaction with other '
281  'Turkers. They have been blocked after being warned '
282  'and failing to adhere. This was done in order to '
283  'ensure a better experience for other '
284  'workers who don\'t disconnect.'
285  )
286  self.mturk_manager.block_worker(worker_id, text)
287  shared_utils.print_and_log(
288  logging.INFO,
289  'Worker {} blocked - too many disconnects'.format(worker_id),
290  True,
291  )
292  elif self.opt['disconnect_qualification'] is not None:
293  self.mturk_manager.soft_block_worker(
294  worker_id, 'disconnect_qualification'
295  )
296  shared_utils.print_and_log(
297  logging.INFO,
298  'Worker {} soft blocked - too many disconnects'.format(
299  worker_id
300  ),
301  True,
302  )
