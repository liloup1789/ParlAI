45 def pad(tensor, length, dim=0, pad=0):
46  """
47  Pad tensor to a specific length.
48 
49  :param tensor: vector to pad
50  :param length: new length
51  :param dim: (default 0) dimension to pad
52 
53  :returns: padded tensor if the tensor is shorter than length
54  """
55  if tensor.size(dim) < length:
56  return torch.cat(
57  [
58  tensor,
59  tensor.new(
60  *tensor.size()[:dim],
61  length - tensor.size(dim),

63  ).fill_(pad),
64  ],
65  dim=dim,
66  )
67  else:
68  return tensor

182  def _decode_forced(self, ys, ctrl_inputs, encoder_states):
183  """
184  Decode with teacher forcing.
185  """
186  bsz = ys.size(0)
187  seqlen = ys.size(1)
188 
189  hidden = encoder_states[1]
190  attn_params = (encoder_states[0], encoder_states[2])
191 
192  # input to model is START + each target except the last
193  y_in = ys.narrow(1, 0, seqlen - 1)
194  xs = torch.cat([self._starts(bsz), y_in], 1)
195 
196  scores = []
197  if self.attn_type == 'none':
198  # do the whole thing in one go
199  output, hidden = self.decoder(xs, ctrl_inputs, hidden, attn_params)
200  score = self.output(output)
201  scores.append(score)
202  else:
203  # need to feed in one token at a time so we can do attention
204  for i in range(seqlen):
205  xi = xs.select(1, i).unsqueeze(1)
206  output, hidden = self.decoder(xi, ctrl_inputs, hidden, attn_params)
207  score = self.output(output)
208  scores.append(score)
209 
210  scores = torch.cat(scores, 1)
211  return scores

213  def _decode(self, ctrl_inputs, encoder_states, maxlen):
214  """
215  Decode maxlen tokens.
216  """
217  hidden = encoder_states[1]
218  attn_params = (encoder_states[0], encoder_states[2])
219  bsz = encoder_states[0].size(0)
220 
221  xs = self._starts(bsz) # input start token
222 
223  scores = []
224  for _ in range(maxlen):
225  # generate at most longest_label tokens
226  output, hidden = self.decoder(xs, ctrl_inputs, hidden, attn_params)
227  score = self.output(output)
228  scores.append(score)
229  xs = score.max(2)[1] # next input is current predicted output
230 
231  scores = torch.cat(scores, 1)
232  return scores

234  def _align_inds(self, encoder_states, cand_inds):
235  """
236  Select the encoder states relevant to valid candidates.
237  """
238  enc_out, hidden, attn_mask = encoder_states
239 
240  # LSTM or GRU/RNN hidden state?
241  if isinstance(hidden, torch.Tensor):
242  hid, cell = hidden, None
243  else:
244  hid, cell = hidden
245 
246  if len(cand_inds) != hid.size(1):
247  # if the number of candidates is mismatched from the number of
248  # hidden states, we throw out the hidden states we won't rank with
249  cand_indices = hid.new(cand_inds)
250  hid = hid.index_select(1, cand_indices)
251  if cell is None:
252  hidden = hid
253  else:
254  cell = cell.index_select(1, cand_indices)
255  hidden = (hid, cell)
256 
257  if self.attn_type != 'none':
258  enc_out = enc_out.index_select(0, cand_indices)
259  attn_mask = attn_mask.index_select(0, cand_indices)
260 
261  return enc_out, hidden, attn_mask

263  def _extract_cur(self, encoder_states, index, num_cands):
264  """
265  Extract encoder states at current index and expand them.
266  """
267  enc_out, hidden, attn_mask = encoder_states
268  if isinstance(hidden, torch.Tensor):
269  cur_hid = hidden.select(1, index).unsqueeze(1).expand(-1, num_cands, -1)
270  else:
271  cur_hid = (
272  hidden[0]
273  .select(1, index)
274  .unsqueeze(1)
275  .expand(-1, num_cands, -1)
276  .contiguous(),
277  hidden[1]
278  .select(1, index)
279  .unsqueeze(1)
280  .expand(-1, num_cands, -1)
281  .contiguous(),
282  )
283 
284  cur_enc, cur_mask = None, None
285  if self.attn_type != 'none':
286  cur_enc = enc_out[index].unsqueeze(0).expand(num_cands, -1, -1)
287  cur_mask = attn_mask[index].unsqueeze(0).expand(num_cands, -1)

290  def _rank(self, cands, cand_inds, encoder_states):
291  """
292  Rank each cand by the average log-probability of the sequence.
293  """
294  if cands is None:
295  return None
296  encoder_states = self._align_inds(encoder_states, cand_inds)
297 
298  cand_scores = []
299  for batch_idx in range(len(cands)):
300  # we do one set of candidates at a time
301  curr_cs = cands[batch_idx]
302  num_cands = curr_cs.size(0)
303 
304  # select just the one hidden state
305  cur_enc_states = self._extract_cur(encoder_states, batch_idx, num_cands)
306 
307  score = self._decode_forced(curr_cs, None, cur_enc_states)
308  true_score = F.log_softmax(score, dim=2).gather(2, curr_cs.unsqueeze(2))
309  nonzero = curr_cs.ne(0).float()
310  scores = (true_score.squeeze(2) * nonzero).sum(1)
311  seqlens = nonzero.sum(1)
312  scores /= seqlens
313  cand_scores.append(scores)
314 
315  max_len = max(len(c) for c in cand_scores)
316  cand_scores = torch.cat(
317  [pad(c, max_len, pad=self.NULL_IDX).unsqueeze(0) for c in cand_scores], 0
318  )
319  return cand_scores

321  def forward(
322  self,
323  xs,
324  ctrl_inputs=None,
325  ys=None,
326  cands=None,
327  prev_enc=None,
328  maxlen=None,
329  seq_len=None,
330  ):
331  """
332  Get output predictions from the model.
333 
334  :param xs:
335  (bsz x seqlen) LongTensor input to the encoder
336  :param ys:
337  expected output from the decoder. used for teacher forcing
338  to calculate loss.
339  :param cands:
340  set of candidates to rank
341  :param prev_enc:
342  if you know you'll pass in the same xs multiple times, you can pass
343  in the encoder output from the last forward pass to skip
344  recalcuating the same encoder output.
345  :param maxlen:
346  max number of tokens to decode. if not set, will use the length of
347  the longest label this model has seen. ignored when ys is not None.
348  :param seq_len:
349  this is the sequence length of the input (xs), i.e. xs.size(1). we
350  use this to recover the proper output sizes in the case when we
351  distribute over multiple gpus
352  :param ctrl_inputs:
353  (bsz x num_controls) LongTensor containing control vars
354 
355  :returns:
356  scores, candidate scores, and encoder states
357 
358  - scores contains the model's predicted token scores.
359  (bsz x seqlen x num_features)
360  - candidate scores are the score the model assigned to each candidate
361  (bsz x num_cands)
362  - encoder states are the (output, hidden, attn_mask) states from the
363  encoder. feed this back in to skip encoding on the next call.
364  """
365  if ys is not None:
366  # keep track of longest label we've ever seen
367  # we'll never produce longer ones than that during prediction
368  self.longest_label = max(self.longest_label, ys.size(1))
369 
370  encoder_states = self._encode(xs, prev_enc)
371 
372  # rank candidates if they are available
373  cand_scores = None
374  if cands is not None:
375  cand_inds = [i for i in range(cands.size(0))]
376  cand_scores = self._rank(cands, cand_inds, encoder_states)
377 
378  if ys is not None:
379  # use teacher forcing
380  scores = self._decode_forced(ys, ctrl_inputs, encoder_states)
381  else:
382  scores = self._decode(
383  ctrl_inputs, encoder_states, maxlen or self.longest_label
384  )
385 
386  if seq_len is not None:
387  # when using multiple gpus, we need to make sure output of
388  # encoder is correct size for gathering; we recover this with
389  # the parameter seq_len
390  if encoder_states[0].size(1) < seq_len:
391  out_pad_tensor = torch.zeros(
392  encoder_states[0].size(0),
393  seq_len - encoder_states[0].size(1),
394  encoder_states[0].size(2),
395  ).cuda()
396  new_out = torch.cat([encoder_states[0], out_pad_tensor], 1)
397  encoder_states = (new_out, encoder_states[1], encoder_states[2])
398 
399  return scores, cand_scores, encoder_states

402 class ControlEncoder(nn.Module):
403  """
404  Given CT control variable, gives the concatenated control embeddings vector.
405  """
406 
407  def __init__(self, control_settings):
408  super().__init__()
409  self.control_settings = control_settings # see ControllableSeq2seqAgent
410 
411  # int to string mapping giving the canonical ordering of the controls
412  self.idx2ctrl = {
413  d['idx']: control for control, d in self.control_settings.items()
414  }
415 
416  # Initialize control embeddings
417  self.control_embeddings = nn.ModuleDict(
418  {
419  c: nn.Embedding(d['num_buckets'], d['embsize'], sparse=False)
420  for c, d in control_settings.items()
421  }
422  ) # maps from string (ctrl name) to nn.Embedding
423 
424  def forward(self, control_inputs):
425  """
426  Forward pass.
427 
428  :param control_inputs: (bsz x num_control_vars) LongTensor of control
429  variable values (i.e. bucket ids)
430 
431  :returns: control_embs, (bsz x sum of control emb sizes) FloatTensor of
432  control variable embeddings, concatenated
433  """
434  # list length num_control_vars of tensors shape (bsz, 1)
435  control_inputs = torch.split(control_inputs, 1, dim=1)
436 
437  # list length num_control_vars of tensors shape (bsz)
438  control_inputs = [torch.squeeze(t, 1) for t in control_inputs]
439 
440  assert len(control_inputs) == len(self.control_settings)
441 
442  control_embs = []
443  for idx, inputs in enumerate(control_inputs):
444  control = self.idx2ctrl[idx] # string, the ctrl var for this position
445  # Append a tensor shape (bsz, control embedding size)
446  control_embs.append(self.control_embeddings[control](inputs))
447  control_embs = torch.cat(control_embs, dim=1) # shape (bsz, sum of ctrl embszs)
448 
449  return control_embs

538  def forward(self, xs):
539  """
540  Encode sequence.
541 
542  :param xs: (bsz x seqlen) LongTensor of input token indices
543 
544  :returns: encoder outputs, hidden state, attention mask
545  encoder outputs are the output state at each step of the encoding.
546  the hidden state is the final hidden state of the encoder.
547  the attention mask is a mask of which input values are nonzero.
548  """
549  bsz = len(xs)
550 
551  # embed input tokens
552  xs = self.input_dropout(xs)
553  xes = self.dropout(self.lt(xs))
554  attn_mask = xs.ne(0)
555  try:
556  x_lens = torch.sum(attn_mask.int(), dim=1)
557  xes = pack_padded_sequence(xes, x_lens, batch_first=True)
558  packed = True
559  except ValueError:
560  # packing failed, don't pack then
561  packed = False
562 
563  encoder_output, hidden = self.rnn(xes)
564  if packed:
565  encoder_output, _ = pad_packed_sequence(encoder_output, batch_first=True)
566  if self.dirs > 1:
567  # project to decoder dimension by taking sum of forward and back
568  if isinstance(self.rnn, nn.LSTM):
569  hidden = (
570  hidden[0].view(-1, self.dirs, bsz, self.hsz).sum(1),
571  hidden[1].view(-1, self.dirs, bsz, self.hsz).sum(1),
572  )
573  else:
574  hidden = hidden.view(-1, self.dirs, bsz, self.hsz).sum(1)
575 
576  return encoder_output, hidden, attn_mask

642  def forward(self, xs, ctrl_inputs=None, hidden=None, attn_params=None):
643  """
644  Decode from input tokens.
645 
646  :param xs: (bsz x seqlen) LongTensor of input token indices
647  :param ctrl_inputs: (bsz, num_controls) LongTensor
648  :param hidden: hidden state to feed into decoder. default (None)
649  initializes tensors using the RNN's defaults.
650  :param attn_params: (optional) tuple containing attention parameters,
651  default AttentionLayer needs encoder_output states
652  and attention mask (e.g. encoder_input.ne(0))
653 
654  :returns: output state(s), hidden state.
655  output state of the encoder. for an RNN, this is
656  (bsz, seq_len, num_directions * hiddensize).
657  hidden state will be same dimensions as input
658  hidden state. for an RNN, this is a tensor of sizes
659  (bsz, numlayers * num_directions, hiddensize).
660  """
661  # sequence indices => sequence embeddings
662  xes = self.dropout(self.lt(xs)) # shape (bsz, seqlen, embsize)
663 
664  # Concatenate the control embeddings
665  if ctrl_inputs is not None:
666  ctrl_embs = self.dropout(
667  self.control_encoder(ctrl_inputs)
668  ) # shape (bsz, sum of ctrl emb sizes)
669  ctrl_embs_tiled = ctrl_embs.unsqueeze(1).repeat(
670  1, xes.size(1), 1
671  ) # shape (bsz, seqlen, sum of ctrl emb sizes)
672  xes = torch.cat(
673  [xes, ctrl_embs_tiled], 2
674  ) # shape (bsz,seqlen,embsize+sum of ctrl embsz)
675 
676  if self.attn_time == 'pre':
677  # modify input vectors with attention
678  xes, _attw = self.attention(xes, hidden, attn_params)
679 
680  # feed tokens into rnn
681  output, new_hidden = self.rnn(xes, hidden)
682 
683  if self.attn_time == 'post':
684  # modify output vectors with attention
685  output, _attw = self.attention(output, new_hidden, attn_params)
686 
687  return output, new_hidden

690 class OutputLayer(nn.Module):
691  """
692  Takes in final states and returns distribution over candidates.
693  """
694 
695  def __init__(
696  self,
697  num_features,
698  embeddingsize,
699  hiddensize,
700  dropout=0,
701  numsoftmax=1,
702  shared_weight=None,
703  padding_idx=-1,
704  ):
705  """
706  Initialize output layer.
707 
708  :param num_features: number of candidates to rank
709  :param hiddensize: (last) dimension of the input vectors
710  :param embeddingsize: (last) dimension of the candidate vectors
711  :param numsoftmax: (default 1) number of softmaxes to calculate.
712  see arxiv.org/abs/1711.03953 for more info.
713  increasing this slows down computation but can
714  add more expressivity to the embeddings.
715  :param shared_weight: (num_features x esz) vector of weights to use as
716  the final linear layer's weight matrix. default
717  None starts with a new linear layer.
718  :param padding_idx: model should output a large negative number for
719  score at this index. if set to -1 (default),
720  this is disabled. if >= 0, subtracts one from
721  num_features and always outputs -1e20 at this
722  index. only used when shared_weight is not None.
723  setting this param helps protect gradient from
724  entering shared embedding matrices.
725  """
726  super().__init__()
727  self.dropout = nn.Dropout(p=dropout)
728 
729  self.padding_idx = padding_idx if shared_weight is not None else -1
730 
731  # embedding to scores
732  if shared_weight is None:
733  # just a regular linear layer
734  self.e2s = nn.Linear(embeddingsize, num_features, bias=True)
735  else:
736  # use shared weights and a bias layer instead
737  if padding_idx == 0:
738  num_features -= 1 # don't include padding
739  shared_weight = shared_weight.narrow(0, 1, num_features)
740  elif padding_idx > 0:
741  raise RuntimeError('nonzero pad_idx not yet implemented')
742  self.weight = Parameter(shared_weight)
743  self.bias = Parameter(torch.Tensor(num_features))
744  self.reset_parameters()
745  self.e2s = lambda x: F.linear(x, self.weight, self.bias)
746 
747  self.numsoftmax = numsoftmax
748  if numsoftmax > 1:
749  self.esz = embeddingsize
750  self.softmax = nn.Softmax(dim=1)
751  self.prior = nn.Linear(hiddensize, numsoftmax, bias=False)
752  self.latent = nn.Linear(hiddensize, numsoftmax * embeddingsize)
753  self.activation = nn.Tanh()
754  else:
755  # rnn output to embedding
756  if hiddensize != embeddingsize:
757  # learn projection to correct dimensions
758  self.o2e = nn.Linear(hiddensize, embeddingsize, bias=True)
759  else:
760  # no need for any transformation here
761  self.o2e = lambda x: x
762 
763  def reset_parameters(self):
764  """
765  Reset bias param.
766  """
767  if hasattr(self, 'bias'):
768  stdv = 1.0 / math.sqrt(self.bias.size(0))
769  self.bias.data.uniform_(-stdv, stdv)
770 
771  def forward(self, input):
772  """
773  Compute scores from inputs.
774 
775  :param input: (bsz x seq_len x num_directions * hiddensize) tensor of
776  states, e.g. the output states of an RNN
777 
778  :returns: (bsz x seqlen x num_cands) scores for each candidate
779  """
780  # next compute scores over dictionary
781  if self.numsoftmax > 1:
782  bsz = input.size(0)
783  seqlen = input.size(1) if input.dim() > 1 else 1
784 
785  # first compute different softmax scores based on input vec
786  # hsz => numsoftmax * esz
787  latent = self.latent(input)
788  active = self.dropout(self.activation(latent))
789  # esz => num_features
790  logit = self.e2s(active.view(-1, self.esz))
791 
792  # calculate priors: distribution over which softmax scores to use
793  # hsz => numsoftmax
794  prior_logit = self.prior(input).view(-1, self.numsoftmax)
795  # softmax over numsoftmax's
796  prior = self.softmax(prior_logit)
797 
798  # now combine priors with logits
799  prob = self.softmax(logit).view(bsz * seqlen, self.numsoftmax, -1)
800  probs = (prob * prior.unsqueeze(2)).sum(1).view(bsz, seqlen, -1)
801  scores = probs.log()
802  else:
803  # hsz => esz, good time for dropout
804  e = self.dropout(self.o2e(input))
805  # esz => num_features
806  scores = self.e2s(e)
807 
808  if self.padding_idx == 0:
809  pad_score = scores.new(scores.size(0), scores.size(1), 1).fill_(-NEAR_INF)
810  scores = torch.cat([pad_score, scores], dim=-1)
811 
812  return scores

815 class AttentionLayer(nn.Module):
816  """
817  Computes attention between hidden and encoder states.
818 
819  See arxiv.org/abs/1508.04025 for more info on each attention type.
820  """
821 
822  def __init__(
823  self,
824  attn_type,
825  hiddensize,
826  embeddingsize,
827  bidirectional=False,
828  attn_length=-1,
829  attn_time='pre',
830  ):
831  """
832  Initialize attention layer.
833  """
834  super().__init__()
835  self.attention = attn_type
836 
837  if self.attention != 'none':
838  hsz = hiddensize
839  hszXdirs = hsz * (2 if bidirectional else 1)
840  if attn_time == 'pre':
841  # attention happens on the input embeddings
842  input_dim = embeddingsize
843  elif attn_time == 'post':
844  # attention happens on the output of the rnn
845  input_dim = hsz
846  else:
847  raise RuntimeError('unsupported attention time')
848 
849  # linear layer for combining applied attention weights with input
850  self.attn_combine = nn.Linear(hszXdirs + input_dim, input_dim, bias=False)
851 
852  if self.attention == 'local':
853  # local attention over fixed set of output states
854  if attn_length < 0:
855  raise RuntimeError('Set attention length to > 0.')
856  self.max_length = attn_length
857  # combines input and previous hidden output layer
858  self.attn = nn.Linear(hsz + input_dim, attn_length, bias=False)
859  # combines attention weights with encoder outputs
860  elif self.attention == 'concat':
861  self.attn = nn.Linear(hsz + hszXdirs, hsz, bias=False)
862  self.attn_v = nn.Linear(hsz, 1, bias=False)
863  elif self.attention == 'general':
864  # equivalent to dot if attn is identity
865  self.attn = nn.Linear(hsz, hszXdirs, bias=False)
866 
867  def forward(self, xes, hidden, attn_params):
868  """
869  Compute attention over attn_params given input and hidden states.
870 
871  :param xes: input state. will be combined with applied
872  attention.
873  :param hidden: hidden state from model. will be used to select
874  states to attend to in from the attn_params.
875  :param attn_params: tuple of encoder output states and a mask showing
876  which input indices are nonzero.
877 
878  :returns: output, attn_weights
879  output is a new state of same size as input state `xes`.
880  attn_weights are the weights given to each state in the
881  encoder outputs.
882  """
883  if self.attention == 'none':
884  # do nothing, no attention
885  return xes, None
886 
887  if type(hidden) == tuple:
888  # for lstms use the "hidden" state not the cell state
889  hidden = hidden[0]
890  last_hidden = hidden[-1] # select hidden state from last RNN layer
891 
892  enc_out, attn_mask = attn_params
893  bsz, seqlen, hszXnumdir = enc_out.size()
894  numlayersXnumdir = last_hidden.size(1)
895 
896  if self.attention == 'local':
897  # local attention weights aren't based on encoder states
898  h_merged = torch.cat((xes.squeeze(1), last_hidden), 1)
899  attn_weights = F.softmax(self.attn(h_merged), dim=1)
900 
901  # adjust state sizes to the fixed window size
902  if seqlen > self.max_length:
903  offset = seqlen - self.max_length
904  enc_out = enc_out.narrow(1, offset, self.max_length)
905  seqlen = self.max_length
906  if attn_weights.size(1) > seqlen:
907  attn_weights = attn_weights.narrow(1, 0, seqlen)
908  else:
909  hid = last_hidden.unsqueeze(1)
910  if self.attention == 'concat':
911  # concat hidden state and encoder outputs
912  hid = hid.expand(bsz, seqlen, numlayersXnumdir)
913  h_merged = torch.cat((enc_out, hid), 2)
914  # then do linear combination of them with activation
915  active = F.tanh(self.attn(h_merged))
916  attn_w_premask = self.attn_v(active).squeeze(2)
917  elif self.attention == 'dot':
918  # dot product between hidden and encoder outputs
919  if numlayersXnumdir != hszXnumdir:
920  # enc_out has two directions, so double hid
921  hid = torch.cat([hid, hid], 2)
922  enc_t = enc_out.transpose(1, 2)
923  attn_w_premask = torch.bmm(hid, enc_t).squeeze(1)
924  elif self.attention == 'general':
925  # before doing dot product, transform hidden state with linear
926  # same as dot if linear is identity
927  hid = self.attn(hid)
928  enc_t = enc_out.transpose(1, 2)
929  attn_w_premask = torch.bmm(hid, enc_t).squeeze(1)
930 
931  # calculate activation scores, apply mask if needed
932  if attn_mask is not None:
933  # remove activation from NULL symbols
934  attn_w_premask.masked_fill_(~attn_mask, -NEAR_INF)
935  attn_weights = F.softmax(attn_w_premask, dim=1)
936 
937  # apply the attention weights to the encoder states
938  attn_applied = torch.bmm(attn_weights.unsqueeze(1), enc_out)
939  # concatenate the input and encoder states
940  merged = torch.cat((xes.squeeze(1), attn_applied.squeeze(1)), 1)
941  # combine them with a linear layer and tanh activation
942  output = torch.tanh(self.attn_combine(merged).unsqueeze(1))
943 
944  return output, attn_weights
