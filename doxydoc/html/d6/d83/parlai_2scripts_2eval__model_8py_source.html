144 def eval_model(opt, print_parser=None):
145  """
146  Evaluates a model.
147 
148  :param opt: tells the evaluation function how to run
149  :param bool print_parser: if provided, prints the options that are set within the
150  model after loading the model
151  :return: the final result of calling report()
152  """
153  random.seed(42)
154  if 'train' in opt['datatype'] and 'evalmode' not in opt['datatype']:
155  raise ValueError(
156  'You should use --datatype train:evalmode if you want to evaluate on '
157  'the training set.'
158  )
159 
160  if opt['save_world_logs'] and not opt['report_filename']:
161  raise RuntimeError(
162  'In order to save model replies, please specify the save path '
163  'with --report-filename'
164  )
165 
166  # load model and possibly print opt
167  agent = create_agent(opt, requireModelExists=True)
168  if print_parser:
169  # show args after loading model
170  print_parser.opt = agent.opt
171  print_parser.print_args()
172 
173  tasks = opt['task'].split(',')
174  reports = []
175  for task in tasks:
176  task_report = _eval_single_world(opt, agent, task)
177  reports.append(task_report)
178 
179  report = aggregate_task_reports(
180  reports, tasks, micro=opt.get('aggregate_micro', True)
181  )
182 
183  # print announcments and report
184  print_announcements(opt)
185  print(
186  '[ Finished evaluating tasks {} using datatype {} ]'.format(
187  tasks, opt.get('datatype', 'N/A')
188  )
189  )
190  print(report)
191  _save_eval_stats(opt, report)
192  return report
