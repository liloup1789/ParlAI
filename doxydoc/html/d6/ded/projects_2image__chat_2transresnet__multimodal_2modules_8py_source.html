314  """

395  def choose_topk(
396  self, idx, encoded, candidates, candidates_encoded, one_cand_set, k
397  ):
398  """
399  Choose top k best responses for a single example.
400 
401  :param idx:
402  idx of example in encoded
403  :param encoded:
404  full matrix of encoded representations (for the whole batch)
405  :param candidates:
406  list of candidates
407  :param candidates_encoded:
408  encoding of the candidates
409  :param one_cand_set:
410  true if there is one set of candidates for each example
411  :param k:
412  how many ranked responses to return
413 
414  :return:
415  ranked list of k responses
416  """
417  encoding = encoded[idx : idx + 1, :]
418  scores = torch.mm(
419  candidates_encoded[idx] if not one_cand_set else candidates_encoded,
420  encoding.transpose(0, 1),
421  )
422  if k >= 1:
423  _, index_top = torch.topk(scores, k, dim=0)
424  else:
425  _, index_top = torch.topk(scores, scores.size(0), dim=0)
426  return [
427  candidates[idx][idx2] if not one_cand_set else candidates[idx2]
428  for idx2 in index_top.unsqueeze(1)
429  ]

431  def get_loss(self, total_encoded, labels_encoded):
432  """
433  Compute loss over batch.
434 
435  :param total_encoded:
436  encoding of the examples
437  :param labels_encoded:
438  encoding of the labels
439 
440  :return:
441  total batch loss, and number of correct examples
442  """
443  loss = None
444  num_correct = None
445  if labels_encoded is not None:
446  dot_products = total_encoded.mm(
447  labels_encoded.t()
448  ) # batch_size * batch_size
449  log_prob = torch.nn.functional.log_softmax(dot_products, dim=1)
450  targets = torch.arange(0, len(total_encoded), dtype=torch.long)
451  if self.use_cuda:
452  targets = targets.cuda()
453  loss = torch.nn.functional.nll_loss(log_prob, targets)
454  num_correct = (log_prob.max(dim=1)[1] == targets).float().sum()
455  return loss, num_correct

504  def __init__(
505  self,
506  n_heads,
507  n_layers,
508  hidden_dim,
509  ffn_size,
510  reduction=True,
511  attention_dropout=0.0,
512  relu_dropout=0.0,
513  learn_positional_embeddings=False,
514  ):
515  super().__init__()
516  self.ffn_size = ffn_size
517  self.n_layers = n_layers
518  self.n_heads = n_heads
519  self.out_dim = hidden_dim
520  self.dim = hidden_dim
521  self.reduction = reduction
522  assert hidden_dim % n_heads == 0, "MM-Combiner dim must be multiple of n_heads"
523  n_positions = 1024
524  self.position_embeddings = nn.Embedding(n_positions, hidden_dim)
525  if not learn_positional_embeddings:
526  create_position_codes(
527  n_positions, hidden_dim, out=self.position_embeddings.weight
528  )
529  else:
530  nn.init.normal_(self.position_embeddings.weight, 0, hidden_dim ** -0.5)
531 
532  self.layers = nn.ModuleList()
533  for _ in range(self.n_layers):
534  self.layers.append(
535  TransformerEncoderLayer(
536  n_heads, hidden_dim, ffn_size, attention_dropout, relu_dropout

540  def forward(self, tensor, mask):
541  """
542  Forward pass.
543 
544  :param tensor:
545  a [bsz, seq_len, hidden_dim] FloatTensor
546  :param mask:
547  a [bsz, seq_len] ByteTensor filled with 1 when inside the sequence and 0 outside.
548 
549  :return:
550  output: a [bsz, hidden_dim] FloatTensor of encodings
551  mask: the same as before
552  """
553  seq_len = tensor.size(1)
554  positions = tensor.new(seq_len).long()
555  positions = torch.arange(seq_len, out=positions).unsqueeze(0)
556  tensor = tensor + self.position_embeddings(positions).expand_as(tensor)
557 
558  tensor *= mask.unsqueeze(-1).float()
559  for i in range(self.n_layers):
560  tensor = self.layers[i](tensor, mask)
561 
562  if self.reduction:
563  divisor = mask.float().sum(dim=1).unsqueeze(-1).clamp(min=1e-20)
564  output = tensor.sum(dim=1) / divisor
565  return output
566  else:
567  output = tensor
568  return output, mask
