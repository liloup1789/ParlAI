222 def remove_first_sv(emb, first_sv):
223  """
224  Projects out the first singular value (first_sv) from the embedding (emb).
225 
226  Inputs:
227  emb: torch Tensor shape (glove_dim)
228  first_sv: torch Tensor shape (glove_dim)
229 
230  Returns:
231  new emb: torch Tensor shape (glove_dim)
232  """
233  # Calculate dot prod of emb and first_sv using torch.mm:
234  # (1, glove_dim) x (glove_dim, 1) -> (1,1) -> float
235  dot_prod = torch.mm(torch.unsqueeze(emb, 0), torch.unsqueeze(first_sv, 1)).item()
236  return emb - first_sv * dot_prod
