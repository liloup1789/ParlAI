201  def _score(self, output, cands):

265  def score_candidates(self, batch, cand_vecs, cand_encs=None):
266  """
267  Score candidates.
268  """
269  # convoluted check that not all memories are empty
270  if (
271  self.opt['use_memories']
272  and batch.memory_vecs is not None
273  and sum(len(m) for m in batch.memory_vecs)
274  ):
275  mems = padded_3d(
276  batch.memory_vecs, use_cuda=self.use_cuda, pad_idx=self.NULL_IDX
277  )
278  else:
279  mems = None
280 
281  if cand_encs is not None:
282  # we pre-encoded the candidates, do not re-encode here
283  cand_vecs = None
284 
285  context_h, cands_h = self.model(xs=batch.text_vec, mems=mems, cands=cand_vecs)
286 
287  if cand_encs is not None:
288  cands_h = cand_encs
289  scores = self._score(context_h, cands_h)
290
