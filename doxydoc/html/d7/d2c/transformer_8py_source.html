196  def _score(self, output, cands):
197  if cands.dim() == 2:
198  return torch.matmul(output, cands.t())
199  elif cands.dim() == 3:
200  return torch.bmm(output.unsqueeze(1), cands.transpose(1, 2)).squeeze(1)
201  else:
202  raise RuntimeError(

260  def score_candidates(self, batch, cand_vecs, cand_encs=None):
261  """
262  Score candidates.
263  """
264  # convoluted check that not all memories are empty
265  if (
266  self.opt['use_memories']
267  and batch.memory_vecs is not None
268  and sum(len(m) for m in batch.memory_vecs)
269  ):
270  mems = padded_3d(
271  batch.memory_vecs, use_cuda=self.use_cuda, pad_idx=self.NULL_IDX
272  )
273  else:
274  mems = None
275 
276  if cand_encs is not None:
277  # we pre-encoded the candidates, do not re-encode here
278  cand_vecs = None
279 
280  context_h, cands_h = self.model(xs=batch.text_vec, mems=mems, cands=cand_vecs)
281 
282  if cand_encs is not None:
283  cands_h = cand_encs
284  scores = self._score(context_h, cands_h)
285 
286  return scores
