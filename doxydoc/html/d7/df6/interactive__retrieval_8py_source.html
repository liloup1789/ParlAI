135  def get_chosen_topic_passages(self, chosen_topic):
136  retrieved_txt_format = []
137  if chosen_topic in self.wiki_map:
138  retrieved_txt = self.wiki_map[chosen_topic]
139  retrieved_txts = retrieved_txt.split('\n')
140 
141  if len(retrieved_txts) > 1:
142  combined = ' '.join(retrieved_txts[2:])
143  sentences = self.sent_tok.tokenize(combined)
144  total = 0
145  for sent in sentences:

147  break
148  if len(sent) > 0:
149  retrieved_txt_format.append(' '.join([chosen_topic, sent]))
150  total += 1
151 
152  if len(retrieved_txt_format) > 0:
153  passages = '\n'.join(retrieved_txt_format)
154  else:
155  passages = ''
156 
157  return passages

159  def get_passages(self, act):
160  """
161  Format passages retrieved by taking the first paragraph of the top
162  `num_retrieved` passages.
163  """
164  retrieved_txt = act.get('text', '')
165  cands = act.get('text_candidates', [])
166  if len(cands) > 0:

171  retrieved_txt_format = []

173  paragraphs = ret_txt.split('\n')
174  if len(paragraphs) > 2:

176  for sent in sentences:

178 
179  if len(retrieved_txt_format) > 0:

186  def retriever_act(self, history):
187  """
188  Combines and formats texts retrieved by the TFIDF retriever for the chosen
189  topic, the last thing the wizard said, and the last thing the apprentice said.
190  """
191  # retrieve on chosen topic
192  chosen_topic_txts = None
193  if self.ret_history.get('chosen_topic'):
194  chosen_topic_txts = self.get_chosen_topic_passages(
195  self.ret_history['chosen_topic']
196  )
197 
198  # retrieve on apprentice
199  apprentice_txts = None
200  if self.ret_history.get('apprentice'):
201  apprentice_act = {
202  'text': self.ret_history['apprentice'],
203  'episode_done': True,
204  }
205  self.retriever.observe(apprentice_act)
206  apprentice_txts = self.get_passages(self.retriever.act())
207 
208  # retrieve on wizard
209  wizard_txts = None
210  if self.ret_history.get('wizard'):
211  wizard_act = {'text': self.ret_history['wizard'], 'episode_done': True}
212  self.retriever.observe(wizard_act)
213  wizard_txts = self.get_passages(self.retriever.act())
214 
215  # combine everything
216  combined_txt = ''
217  if chosen_topic_txts:
218  combined_txt += chosen_topic_txts
219  if wizard_txts:
220  combined_txt += '\n' + wizard_txts
221  if apprentice_txts:
222  combined_txt += '\n' + apprentice_txts
223 
224  return combined_txt
