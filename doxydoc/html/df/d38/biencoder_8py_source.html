42  def _build_candidates(self, batch, source, mode):
43  cands, cand_vecs, label_inds = super()._build_candidates(batch, source, mode)
44  label_vecs = batch.label_vec # [bsz] list of lists of LongTensors
45  batchsize = (
46  batch.text_vec.size(0)
47  if batch.text_vec is not None
48  else batch.image.size(0)
49  )
50 
51  if source == 'fixed' and label_inds is None and self.add_label_to_fixed_cands:
52  # Add label to fixed cands
53  if label_vecs is not None:
54  label_inds = label_vecs.new_empty((batchsize))
55  for batch_idx, label_vec in enumerate(label_vecs):
56  max_c_len = cand_vecs.size(1)
57  label_vec_pad = label_vec.new_zeros(max_c_len).fill_(self.NULL_IDX)
58  if max_c_len < len(label_vec):
59  label_vec = label_vec[0:max_c_len]
60  label_vec_pad[0 : label_vec.size(0)] = label_vec
61  label_inds[batch_idx] = self._find_match(cand_vecs, label_vec_pad)
62  if label_inds[batch_idx] == -1:
63  cand_vecs = torch.cat((cand_vecs, label_vec_pad.unsqueeze(0)))
64  cands.append(batch.labels[batch_idx])
65  label_inds[batch_idx] = len(cands) - 1
