Detailed Description

Loads a pre-weighted inverted index of token/document terms.

Scores new queries by taking sparse dot products.

98  def text2spvec(self, query):
99  """
100  Create a sparse tfidf-weighted word vector from query.
101 
102  tfidf = log(tf + 1) * log((N - Nt + 0.5) / (Nt + 0.5))
103  """
104  # Get hashed ngrams
105  words = self.parse(utils.normalize(query))
106  wids = [utils.hash(w, self.hash_size) for w in words]
107 
108  if len(wids) == 0:
109  if self.strict:
110  raise RuntimeError('No valid word in: %s' % query)
111  else:
112  logger.warning('No valid word in: %s' % query)
113  return sp.csr_matrix((1, self.hash_size))
114 
115  # Count TF
116  wids_unique, wids_counts = np.unique(wids, return_counts=True)
117  tfs = np.log1p(wids_counts)
118 
119  # Count IDF
120  Ns = self.doc_freqs[wids_unique]
121  idfs = np.log((self.num_docs - Ns + 0.5) / (Ns + 0.5))
122  idfs[idfs < 0] = 0
123 
124  # TF-IDF
125  data = np.multiply(tfs, idfs)
126 
127  # One row, sparse csr matrix
128  indptr = np.array([0, len(wids_unique)])
129  spvec = sp.csr_matrix((data, wids_unique, indptr), shape=(1, self.hash_size))
130 
131  return spvec
