17 def universal_sentence_embedding(sentences, mask, sqrt=True):
18  """
19  Perform Universal Sentence Encoder averaging (https://arxiv.org/abs/1803.11175).
20 
21  This is really just sum / sqrt(len).
22 
23  :param Tensor sentences: an N x T x D of Transformer outputs. Note this is
24  the exact output of TransformerEncoder, but has the time axis first
25  :param ByteTensor: an N x T binary matrix of paddings
26 
27  :return: an N x D matrix of sentence embeddings
28  :rtype Tensor:
29  """
30  # need to mask out the padded chars
31  sentence_sums = th.bmm(
32  sentences.permute(0, 2, 1), mask.float().unsqueeze(-1)
33  ).squeeze(-1)
34  divisor = mask.sum(dim=1).view(-1, 1).float()
35  if sqrt:
36  divisor = divisor.sqrt()
37  sentence_sums /= divisor
38  return sentence_sums

73 class ContextKnowledgeEncoder(nn.Module):
74  def __init__(self, transformer):
75  super().__init__()
76  # The transformer takes care of most of the work, but other modules
77  # expect us to have an embeddings available
78  self.embeddings = transformer.embeddings
79  self.embed_dim = transformer.embeddings.embedding_dim
80  self.transformer = transformer

82  def forward(self, src_tokens, know_tokens, ck_mask, cs_ids, use_cs_ids):
83  # encode the context, pretty basic
84  context_encoded, context_mask = self.transformer(src_tokens)
85 
86  # make all the knowledge into a 2D matrix to encode
87  N, K, Tk = know_tokens.size()
88  know_flat = know_tokens.reshape(-1, Tk)
89  know_encoded, know_mask = self.transformer(know_flat)
90 
91  # compute our sentence embeddings for context and knowledge
92  context_use = universal_sentence_embedding(context_encoded, context_mask)
93  know_use = universal_sentence_embedding(know_encoded, know_mask)
94 
95  # remash it back into the shape we need
96  know_use = know_use.reshape(N, know_tokens.size(1), self.embed_dim)
97  context_use /= np.sqrt(self.embed_dim)
98  know_use /= np.sqrt(self.embed_dim)
99 
100  ck_attn = th.bmm(know_use, context_use.unsqueeze(-1)).squeeze(-1)
101  # fill with near -inf
102  ck_attn.masked_fill_(~ck_mask, neginf(context_encoded.dtype))
103 
104  if not use_cs_ids:
105  # if we're not given the true chosen_sentence (test time), pick our
106  # best guess
107  _, cs_ids = ck_attn.max(1)
108 
109  # pick the true chosen sentence. remember that TransformerEncoder outputs
110  # (batch, time, embed)
111  # but because know_encoded is a flattened, it's really
112  # (N * K, T, D)
113  # We need to compute the offsets of the chosen_sentences
114  cs_offsets = th.arange(N, device=cs_ids.device) * K + cs_ids
115  cs_encoded = know_encoded[cs_offsets]
116  # but padding is (N * K, T)
117  cs_mask = know_mask[cs_offsets]
118 
119  # finally, concatenate it all
120  full_enc = th.cat([cs_encoded, context_encoded], dim=1)
121  full_mask = th.cat([cs_mask, context_mask], dim=1)
122 
123  # also return the knowledge selection mask for the loss
124  return full_enc, full_mask, ck_attn
