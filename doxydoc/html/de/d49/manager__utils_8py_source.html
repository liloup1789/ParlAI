49  def _safe_send(self, data, force=False):
50  if not self.alive and not force:
51  # Try to wait a second to send a packet
52  timeout = 1
53  while timeout > 0 and not self.alive:
54  time.sleep(0.1)
55  timeout -= 0.1
56  if not self.alive:
57  # don't try to send a packet if we're still dead
58  return False
59  try:
60  self.ws.send(data)
61  except websocket.WebSocketConnectionClosedException:
62  # The channel died mid-send, wait for it to come back up
63  return False
64  return True

125  def on_message(*args):
126  """
127  Incoming message handler for messages from the FB user.
128  """
129  packet_dict = json.loads(args[1])
130  if packet_dict['type'] == 'conn_success':
131  self.alive = True
132  return # No action for successful connection
133  if packet_dict['type'] == 'pong':
134  self.last_pong = time.time()
135  return # No further action for pongs
136  message_data = packet_dict['content']
137  shared_utils.print_and_log(
138  logging.DEBUG, 'Message data received: {}'.format(message_data)
139  )
140  for message_packet in message_data['entry']:
141  for message in message_packet['messaging']:
142  self.message_callback(message)
