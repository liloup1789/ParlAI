95 class retry(object):
96  """
97  Decorator for flaky tests. Test is run up to ntries times, retrying on failure.
98 
99  :param ntries:
100  the number of tries to attempt
101  :param log_retry:
102  if True, prints to stdout on retry to avoid being seen as "hanging"
103 
104  On the last time, the test will simply fail.
105 
106  >>> @retry(ntries=10)
107  ... def test_flaky(self):
108  ... import random
109  ... self.assertLess(0.5, random.random())
110  """
111 
112  def __init__(self, ntries=3, log_retry=False):
113  self.ntries = ntries
114  self.log_retry = log_retry
115 
116  def __call__(self, testfn):
117  """
118  Call testfn(), possibly multiple times on failureException.
119  """
120  from functools import wraps
121 
122  @wraps(testfn)
123  def _wrapper(testself, *args, **kwargs):
124  for _ in range(self.ntries - 1):
125  try:
126  return testfn(testself, *args, **kwargs)
127  except testself.failureException:
128  if self.log_retry:
129  print("Retrying {}".format(testfn))
130  # last time, actually throw any errors there may be
131  return testfn(testself, *args, **kwargs)
132 
133  return _wrapper

249 def timeout(time: int = 30):
250  """
251  Raise a timeout if a function does not return in time `time`.
252 
253  Use as a context manager, so that the signal class can reset it's alarm for
254  `SIGALARM`
255 
256  :param int time:
257  Time in seconds to wait for timeout. Default is 30 seconds.
258  """
259  assert time >= 0, 'Time specified in timeout must be nonnegative.'
260 
261  def _handler(signum, frame):
262  raise TimeoutError
