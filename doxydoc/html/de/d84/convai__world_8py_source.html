212  def pull_new_messages(self):
213  """
214  Requests the server for new messages and processes every message. If a message
215  starts with '/start' string then a new chat will be created and the message will
216  be added to stack. If a message has the same chat id as already existing chat
217  then it will be added to message stack for this chat. Any other messages will be
218  ignored. If after processing all messages message stack is still empty then new
219  request to server will be performed.
220 
221  :return: None
222  """
223  print('Waiting for new messages from server...', flush=True)
224  while True:
225  time.sleep(self.router_bot_pull_delay)
226  msgs = self._get_updates()
227  if len(msgs["result"]) > 0:
228  for msg in msgs["result"]:
229  print('\nProceed message: %s' % msg)
230  text = self._get_message_text(msg)
231  chatID = self._get_chat_id(msg)
232 
233  if self.chats.get(chatID, None) is not None:
234  print('Message was recognized as part of chat #%s' % chatID)
235  self.messages.append((chatID, text))
236  elif self._is_begin_of_conversation(text):
237  print(
238  'Message was recognised as start of new chat #%s' % chatID
239  )
240  if self.bot_capacity == -1 or 0 <= self.bot_capacity > (
241  len(self.chats) - len(self.finished_chats)
242  ):
243  self._init_chat(chatID)
244  text = self._strip_start_message(text)
245  self.messages.append((chatID, text))
246  print(
247  'New world and agents for chat #%s are created.'
248  % chatID
249  )
250  else:
251  print(
252  'Cannot start new chat #%s due to bot capacity'
253  'limit reached.' % chatID
254  )
255  else:
256  print(
257  'Message was not recognized as part of any chat.'
258  'Message skipped.'
259  )
260  if len(self.messages) > 0:
261  break
262  else:
263  print('Waiting for new messages from server...', flush=True)

265  def parley(self):
266  """
267  Pops next message from stack, gets corresponding chat, agents, world and
268  performs communication between agents. Result of communication will be send to
269  server. If message stack is empty then server will be requested for new
270  messages.
271 
272  :return: None
273  """
274  print('Try to cleanup finished chat before new parley.')
275  self.cleanup_finished_chats()
276 
277  if len(self.messages) == 0:
278  print('Message stack is empty. Try to request new messages from server.')
279  self.pull_new_messages()
280 
281  print('Pop next message from stack')
282 
283  (chatID, text) = self.messages.pop(0)
284  episode_done = self._is_end_of_conversation(text)
285  local_agent = self.chats.get(chatID, None)
286 
287  if local_agent is not None:
288  self.curr_chatID = chatID
289  msg = {
290  'id': 'MasterBot#%s' % chatID,
291  'text': text,
292  'episode_done': episode_done,
293  }
294  local_agent.observe(msg)
295  reply = local_agent.act()
296  self.last_exchanges[chatID] = [msg, reply]
297  if self._is_end_of_conversation(reply['text']) or reply['episode_done']:
298  episode_done = True
299 
300  if self._is_skip_response(reply['text']):
301  print('Skip response from agent for chat #%s' % chatID)
302  else:
303  print('Send response from agent to chat #%s: %s' % (chatID, reply))
304  self._send_message(reply, chatID)
305  else:
306  print('Message was not recognized as part of any chat. Message skipped.')
307 
308  if episode_done:
309  self.finished_chats.add(chatID)
