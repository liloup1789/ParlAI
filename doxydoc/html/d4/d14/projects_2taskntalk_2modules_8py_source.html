12 def xavier_init(module):
13  """
14  Xavier initializer for module parameters.
15  """
16  for parameter in module.parameters():
17  if len(parameter.data.shape) == 1:
18  # 1D vector means bias
19  parameter.data.fill_(0)
20  else:
21  fan_in = parameter.data.size(0)
22  fan_out = parameter.data.size(1)
23  parameter.data.normal_(0, math.sqrt(2 / (fan_in + fan_out)))
