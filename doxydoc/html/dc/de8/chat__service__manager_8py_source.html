531  def check_timeout_in_pool(
532  self, world_type, agent_pool, max_time_in_pool, backup_task=None
533  ):
534  """
535  Check for timed-out agents in pool.
536 
537  :param world_type:
538  string world type
539  :param agent_pool:
540  list of ``AgentState``s
541  :param max_time_in_pool:
542  int maximum time allowed for agent to be in pool
543  :param backup_task:
544  string backup_task to start if we reach a timeout in the original pool
545  """
546  for agent_state in agent_pool:
547  time_in_pool = agent_state.time_in_pool.get(world_type)
548  if time_in_pool and time.time() - time_in_pool > max_time_in_pool:
549  # remove agent from agent_pool
550  self.remove_agent_from_pool(agent_state, world_type)
551  # put agent back in overworld
552  agent_state.set_active_agent(agent_state.get_overworld_agent())
553 
554  agent_state.stored_data['removed_after_timeout'] = True
555  self.after_agent_removed(agent_state.service_id)
556 
557  if backup_task is not None:
558  self.add_agent_to_pool(agent_state, backup_task)
559 
560  # reset wait message state
561  agent_state.stored_data['seen_wait_message'] = False
562 
563  elif time_in_pool and time.time() - time_in_pool > 30:
564  # tell agent that a match is taking longer than
565  # expected
566  if (
567  not agent_state.stored_data.get('seen_wait_message')
568  or not agent_state.stored_data['seen_wait_message']
569  ):
570  self.observe_message(
571  agent_state.service_id,
572  'Pairing is taking longer than expected. '
573  'If you wish to exit, type *EXIT*.',
574  )
575  self.message_sender.typing_on(agent_state.service_id)
576  agent_state.stored_data['seen_wait_message'] = True
