34 class Encoder(nn.Module):
35  """
36  Example encoder, consisting of an embedding layer and a 1-layer LSTM with the
37  specified hidden size.
38 
39  Pay particular attention to the ``forward`` output.
40  """
41 
42  def __init__(self, embeddings, hidden_size):
43  """
44  Initialization.
45 
46  Arguments here can be used to provide hyperparameters.
47  """
48  # must call super on all nn.Modules.
49  super().__init__()
50 
51  _vocab_size, esz = embeddings.weight.shape
52  self.embeddings = embeddings
53  self.lstm = nn.LSTM(
54  input_size=esz, hidden_size=hidden_size, num_layers=1, batch_first=True
55  )
56 
57  def forward(self, input_tokens):
58  """
59  Perform the forward pass for the encoder.
60 
61  Input *must* be input_tokens, which are the context tokens given
62  as a matrix of lookup IDs.
63 
64  :param input_tokens:
65  Input tokens as a bsz x seqlen LongTensor.
66  Likely will contain padding.
67 
68  :return:
69  You can return anything you like; it is will be passed verbatim
70  into the decoder for conditioning. However, it should be something
71  you can easily manipulate in ``reorder_encoder_states``.
72  This particular implementation returns the hidden and cell states from the
73  LSTM.
74  """
75  embedded = self.embeddings(input_tokens)
76  _output, hidden = self.lstm(embedded)
77  return hidden

80 class Decoder(nn.Module):
81  """
82  Basic example decoder, consisting of an embedding layer and a 1-layer LSTM with the
83  specified hidden size. Decoder allows for incremental decoding by ingesting the
84  current incremental state on each forward pass.
85 
86  Pay particular note to the ``forward``.
87  """
88 
89  def __init__(self, embeddings, hidden_size):
90  """
91  Initialization.
92 
93  Arguments here can be used to provide hyperparameters.
94  """
95  super().__init__()
96  _vocab_size, self.esz = embeddings.weight.shape
97  self.embeddings = embeddings
98  self.lstm = nn.LSTM(
99  input_size=self.esz, hidden_size=hidden_size, num_layers=1, batch_first=True
100  )
101 
102  def forward(self, input, encoder_state, incr_state=None):
103  """
104  Run forward pass.
105 
106  :param input:
107  The currently generated tokens from the decoder.
108  :param encoder_state:
109  The output from the encoder module.
110  :parm incr_state:
111  The previous hidden state of the decoder.
112  """
113  embedded = self.embeddings(input)
114  if incr_state is None:
115  # this is our very first call. We want to seed the LSTM with the
116  # hidden state of the decoder
117  state = encoder_state
118  else:
119  # We've generated some tokens already, so we can reuse the existing
120  # decoder state
121  state = incr_state
122 
123  # get the new output and decoder incremental state
124  output, incr_state = self.lstm(embedded, state)
125 
126  return output, incr_state

129 class ExampleModel(tga.TorchGeneratorModel):
130  """
131  ExampleModel implements the abstract methods of TorchGeneratorModel to define how to
132  re-order encoder states and decoder incremental states.
133 
134  It also instantiates the embedding table, encoder, and decoder, and defines the
135  final output layer.
136  """
137 
138  def __init__(self, dictionary, esz=256, hidden_size=1024):
139  super().__init__(
140  padding_idx=dictionary[dictionary.null_token],
141  start_idx=dictionary[dictionary.start_token],
142  end_idx=dictionary[dictionary.end_token],
143  unknown_idx=dictionary[dictionary.unk_token],
144  )
145  self.embeddings = nn.Embedding(len(dictionary), esz)
146  self.encoder = Encoder(self.embeddings, hidden_size)
147  self.decoder = Decoder(self.embeddings, hidden_size)
148 
149  def output(self, decoder_output):
150  """
151  Perform the final output -> logits transformation.
152  """
153  return F.linear(decoder_output, self.embeddings.weight)
154 
155  def reorder_encoder_states(self, encoder_states, indices):
156  """
157  Reorder the encoder states to select only the given batch indices.
158 
159  Since encoder_state can be arbitrary, you must implement this yourself.
160  Typically you will just want to index select on the batch dimension.
161  """
162  h, c = encoder_states
163  return h[:, indices, :], c[:, indices, :]
164 
165  def reorder_decoder_incremental_state(self, incr_state, indices):
166  """
167  Reorder the decoder states to select only the given batch indices.
168 
169  This method can be a stub which always returns None; this will result in the
170  decoder doing a complete forward pass for every single token, making generation
171  O(n^2). However, if any state can be cached, then this method should be
172  implemented to reduce the generation complexity to O(n).
173  """
174  h, c = incr_state
175  return h[:, indices, :], c[:, indices, :]
