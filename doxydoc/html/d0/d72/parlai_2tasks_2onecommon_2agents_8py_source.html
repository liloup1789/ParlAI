192  sentences = []
193  spans = []

227  def _split_referents(self, raw_referents, spans):
228  """
229  Split the referents.
230 
231  The first 3 values are begin idx, end idx, and eos idx The next N_OBJECT values
232  are booleans of if the object is referred e.g. 3 4 10 0 1 0 0 0 0 0 means idx 3
233  to 4 is a markable of an utterance with <eos> at idx 10, and it refers to the
234  2nd dot
235  """
236 
237  referent_len = 3 + N_OBJECT
238  splitted_referents = []
239  for i in range(len(raw_referents) // referent_len):
240  val = raw_referents[i * referent_len : (i + 1) * referent_len]
241  splitted_referents.append(list(map(int, val)))
242 
243  referents = []
244  idx = 0
245  for span in spans:
246  if span is None:
247  referents.append(None)
248  continue
249 
250  # span is a (bos index, eos index) of an utterance
251  refs = []
252  while idx < len(splitted_referents):
253  if splitted_referents[idx][REF_EOS_IDX] == span[1]:
254  ref = {
255  'begin': splitted_referents[idx][REF_BEGIN_IDX] - (span[0] + 1),
256  'end': splitted_referents[idx][REF_END_IDX] - (span[0] + 1),
257  'target': splitted_referents[idx][REF_BEGIN_TARGET_IDX:],
258  }
259  refs.append(ref)
260  idx += 1
261  else:
262  break
263  referents.append(refs)
264 
265  return referents
