19 class SharedTable(MutableMapping):
20  """
21  Provides a simple shared-memory table of integers, floats, or strings.
22 
23  Use this class as follows:
24 
25  .. code-block:: python
26 
27  tbl = SharedTable({'cnt': 0})
28  with tbl.get_lock():
29  tbl['startTime'] = time.time()
30  for i in range(10):
31  with tbl.get_lock():
32  tbl['cnt'] += 1
33  """
34 
35  types = {int: ctypes.c_int, float: ctypes.c_float, bool: ctypes.c_bool}
36 
37  def __init__(self, init_dict=None):
38  """
39  Create a shared memory version of each element of the initial dictionary.
40 
41  Creates an empty array otherwise, which will extend
42  automatically when keys are added.
43 
44  Each different type (all supported types listed in the ``types`` array
45  above) has its own array. For each key we store an index into the
46  appropriate array as well as the type of value stored for that key.
47  """
48  # idx is dict of {key: (array_idx, value_type)}
49  self.idx = {}
50  # arrays is dict of {value_type: array_of_ctype}
51  self.arrays = {}
52  self.tensors = {}
53 
54  if init_dict:
55  sizes = {typ: 0 for typ in self.types.keys()}
56  for k, v in init_dict.items():
57  if is_tensor(v):
58  # add tensor to tensor dict--don't try to put in rawarray
59  self.tensors[k] = v
60  continue
61  elif type(v) not in sizes:
62  raise TypeError(
63  'SharedTable does not support values of '
64  + 'type '
65  + str(type(v))
66  )
67  sizes[type(v)] += 1
68  # pop tensors from init_dict
69  for k in self.tensors.keys():
70  init_dict.pop(k)
71  # create raw arrays for each type
72  for typ, sz in sizes.items():
73  self.arrays[typ] = RawArray(self.types[typ], sz)
74  # track indices for each key, assign them to their typed rawarray
75  idxs = {typ: 0 for typ in self.types.keys()}
76  for k, v in init_dict.items():
77  val_type = type(v)
78  self.idx[k] = (idxs[val_type], val_type)
79  if val_type == str:
80  v = sys.intern(v)
81  self.arrays[val_type][idxs[val_type]] = v
82  idxs[val_type] += 1
83  # initialize any needed empty arrays
84  for typ, ctyp in self.types.items():
85  if typ not in self.arrays:
86  self.arrays[typ] = RawArray(ctyp, 0)
87  self.lock = Lock()

110  def __setitem__(self, key, value):
111  """
112  If key is in table, update it. Otherwise, extend the array to make room.
113 
114  This uses additive resizing not multiplicative, since the number
115  of keys is not likely to change frequently during a run, so do not
116  abuse it.
117 
118  Raises an error if you try to change the type of the value stored for
119  that key -- if you need to do this, you must delete the key first.
120  """
121  val_type = type(value)
122  if 'Tensor' in str(val_type):
123  self.tensors[key] = value
124  return
125  if val_type not in self.types:
126  raise TypeError('SharedTable does not support type ' + str(type(value)))
127  if val_type == str:
128  value = sys.intern(value)
129  if key in self.idx:
130  idx, typ = self.idx[key]
131  if typ != val_type:
132  raise TypeError(
133  (
134  'Cannot change stored type for {key} from '
135  + '{v1} to {v2}. You need to del the key first'
136  + ' if you need to change value types.'
137  ).format(key=key, v1=typ, v2=val_type)
138  )
139  self.arrays[typ][idx] = value
140  else:
141  raise KeyError(
142  'Cannot add more keys to the shared table as '
143  'they will not be synced across processes.'
144  )
